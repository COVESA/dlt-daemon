include_directories(
    ${PROJECT_SOURCE_DIR}/
    ${PROJECT_SOURCE_DIR}/include/dlt
    ${DBUS_INCLUDE_DIRS}

	DESTINATION ${CONFIGURATION_FILES_DIR}
	COMPONENT base)
