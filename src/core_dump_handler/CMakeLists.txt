    set(COREDUMP_CONF_DIR "/usr/lib/sysctl.d/")

    #add_definitions( -fno-strict-aliasing )

    install(FILES ${PROJECT_BINARY_DIR}/core_dump_handler/50-coredump.conf DESTINATION ${COREDUMP_CONF_DIR})
